const box2 = document.querySelector("#section2 .parallax__item__img");
const box3 = document.querySelector("#section3 .parallax__item__img");
const box4 = document.querySelector("#section4 .parallax__item__img");
const box5 = document.querySelector("#section5 .parallax__item__img");
const box6 = document.querySelector("#section6 .parallax__item__img");

// 01 : 기본 애니메이션
gsap.to(box1, {
    duration: 2,            // 애니메이션 지속 시간
    x: 500,                 // x 좌표 이동
    borderRadius : 100,     // border-radius 속성 변경
    rotation: 360,          // 회전
});

// 02 : ScrollTrigger를 사용한 애니메이션
gsap.to(box2, {
    duration: 2,
    x: 500,
    rotation: 360,
    borderRadius: 100,

    scrollTrigger: {
        trigger: box2       // 애니메이션 트리거 요소
    }
});

// 03 : toggleActions를 사용한 애니메이션
gsap.to(box3,{
    duration: 2,
    x: 500,
    rotation: 360,
    borderRadius: 100,

    scrollTrigger: {
        trigger: box3,
        toggleActions: 'play none reverse none' // 트리거 진입, 트리거 나감, 트리거 되돌아옴, 트리거 되돌아옴 상태에서 수행할 동작 설정
                                                // 종류 play pause resume reverse restart reset complete none
                                                // 종류 onEnter onLeave onEnterBack onLeaveBack
    }
});

// 04 : start, end를 사용한 애니메이션
gsap.to(box4,{
    duration: 2,
    x: 500,
    rotation: 360,
    borderRadius: 100,

    scrollTrigger: {
        trigger: box4,
        start: "top 50%",                           // 시작 위치 설정
        end: "bottom 20%",                          // 종료 위치 설정
        toggleActions:"play none reverse none",
        markers: false
    }
});

// 05 : scrub를 사용한 애니메이션
gsap.to(box5,{
    duration: 2,
    x: 500,
    rotation: 360,
    borderRadius: 100,

    scrollTrigger: {
        trigger: box5,
        start: "top 50%",
        end: "bottom 20%",
        scrub: true                             // ScrollTrigger 스크럽(Scrubbing) 활성화
    }
});

// 06 : pin을 사용한 애니메이션
gsap.to(box6,{
    duration: 2,
    x: 500,
    rotation: 360,
    borderRadius: 100,

    scrollTrigger: {
        trigger: box6,
        start: "top 50%",
        end: "bottom 400px",    // pin이 트리거 요소에서 벗어날 때 설정
        scrub: true,
        pin: true               // ScrollTrigger 항목을 고정
    }
});

// 07 : toggleClass를 사용한 애니메이션
gsap.to(box7,{
    duration: 2,
    x: 500,
    rotation: 360,
    borderRadius: 100,

    scrollTrigger: {
        trigger: box7,
        start: "top center",
        end: "bottom 20%",
        scrub: true,
        toggleClass: "active",          // 트리거 특정 범위 내에서 클래스 토글
        id: "box7",                     // ScrollTrigger 식별자
        markers: true
    }
});

// 08 : callback 함수를 사용한 애니메이션
gsap.to(box8,{
    duration: 2,
    x: 500,
    rotation: 360,
    borderRadius: 100,

    scrollTrigger: {
        trigger: box8,
        start: "top center",
        end: "bottom 20%",
        scrub: true,
        // onEnter: () => {console.log("onEnter")},                     // 트리거 진입 시 콜백 함수
        // onLeave: () => {console.log("onLeave")},                     // 트리거 나감 시 콜백 함수
        // onEnterBack: () => {console.log("onEnterBack")},             // 트리거 되돌아옴 시 콜백 함수
        // onLeaveBack: () => {console.log("onLeaveBack")},             // 트리거 되돌아옴 상태에서 콜백 함수
        onUpdate: (self) => {console.log("onUpdate", self.progress.toFixed(3))},        // 애니메이션 진행률 업데이트 시 콜백 함수(0부터 1까지의 값으로 애니메이션 진행률 표시, 진행률을 소수 셋째 자리까시 표시)
        onToggle: (self) => {console.log("onToggle", self.isActive)}                    // 트리거 토글 시 콜백 함수(self.isActive는 ScrollTrigger가 활성화되었을 때 true를 반환, 비활성화되었을 때 false를 반환)
    }
});
